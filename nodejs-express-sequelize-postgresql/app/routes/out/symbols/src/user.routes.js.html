1 /**
  2  *  Defining routes to determine how the server will response
  3  *  When a client sends request for an endpoint using HTTP request
  4  *  @author nandorusrin
  5  *
  6  */
  7 
  8 module.exports = app => {
  9   const users = require("../controllers/user.controller.js");
 10 
 11   var router = require("express").Router();
 12 
 13   /** Create a new Tutorial */
 14   router.post("/", users.create);
 15 
 16   /** Retrieve all Tutorials */
 17   router.get("/", users.findAll);
 18 
 19   /** Retrieve a single Tutorial with id */
 20   router.get("/:id", users.findOne);
 21 
 22   /** Put a Tutorial with id */
 23   router.put("/:id", users.updateAll);
 24 
 25   /** Patch a Tutorial with id */
 26   router.patch("/:id", users.update);
 27 
 28   /** Softdelete a Tutorial with id, by adding deletedAt */
 29   router.delete("/:id", users.delete);
 30 
 31   app.use('/api/v1/users', router);
 32 };
