  7 const db = require("../models");
  8 const User = db.users;
  9 const Op = db.Sequelize.Op;

 11 /** Create and Save a new User */
 12 exports.create = (req, res) => {
 13   /** Validate  */
 14   var rgxNoNumber = /\d/g;
 15   if (rgxNoNumber.test(req.body.name)) {
 16     res.status(400).send({
 17       message: "Content name can not contain number!"
 18     });
 19     return;
 20   }
 21 
 22   var rgxOnlyNumber = /^\d+$/;
 23   if (!rgxOnlyNumber.test(req.body.indonesianID) && req.body.indonesianID.length == 16) {
 24     res.status(400).send({
 25       message: "Content ID can only contain number and 16 in length!"
 26     });
 27     return;
 28   }
 29 
 30   /** Create a User */
 31   const user = {
 32     name: req.body.name,
 33     indonesianID: req.body.indonesianID,
 34     birthday: req.body.birthday,
 35     deletedAt: req.body.deletedAt
 36   };
 37 
 38   /** Save User in the database */
 39   User.create(user)
 40     .then(data => {
 41       res.send(data);
 42     })
 43     .catch(err => {
 44       res.status(500).send({
 45         message:
 46           err.message || "Some error occurred while creating the User."
 47       });
 48     });
 49 };

 51 /** Retrieve all Users from the database. */
 52 exports.findAll = (req, res) => {
 53   const name = req.query.name;
 54   var condition = name ? { name: { [Op.iLike]: `%${name}%` } } : null;
 55 
 56   User.findAll({ where: condition }, { offset: 3, limit: 1 })

 59     })
 60     .catch(err => {
