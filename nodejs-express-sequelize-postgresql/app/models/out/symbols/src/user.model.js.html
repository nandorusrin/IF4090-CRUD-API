1 /**
  2  *  Defining the Sequelize model
  3  *  Sequelize Model represents users table in PostgreSQL database. 
  4  *  These columns will be generated automatically: id, name, indonesianID, birthday, deletedAt, createdAt, updatedAt.
  5  *  @author nandorusrin
  6  *
  7  */
  8 
  9 module.exports = (sequelize, Sequelize) => {
 10   const User = sequelize.define("user", {
 11     name: {
 12       type: Sequelize.STRING
 13     },
 14     indonesianID: {
 15       type: Sequelize.STRING
 16     },
 17     birthday: {
 18       type: Sequelize.DATE
 19     },
 20     deletedAt: {
 21       type: Sequelize.DATE
 22     }
 23   });
 24 
 25   return User;
 26 };
